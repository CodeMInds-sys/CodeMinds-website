اختبار C++ OOP

1. ما ناتج تنفيذ الكود التالي؟

class MyClass {
private:
    int x;

public:
    void setX(int val) {
        x = val;
    }

    int getX() {
        return x;
    }
};

int main() {
    MyClass obj;
    obj.setX(10);
    cout << obj.getX();
}



2. ما الإخراج المتوقع؟

class Test {
public:
    Test() { cout << "Default constructor\n"; }
    Test(int x) { cout << "Parameterized constructor\n"; }
};

int main() {
    Test a;
    Test b(5);
}



3. ما الذي سيتم طباعته؟

class Base {
public:
    void show() { cout << "Base\n"; }
};

class Derived : public Base {
public:
    void show() { cout << "Derived\n"; }
};

int main() {
    Derived d;
    d.show();
}



4. ما ناتج الطباعة؟

class Counter {
public:
    static int count;
    Counter() { count++; }
};

int Counter::count = 0;

int main() {
    Counter a, b, c;
    cout << Counter::count;
}
